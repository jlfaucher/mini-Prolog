 102    type Type_Equation is (Libre,                             -- La variable est libre.
 103       Gele,                              -- La variable est rattachée à un processus gelé.
 104       Lie,                               -- La variable est liée à un objet PROLOG (elle a une valeur).
 105       Ref);                              -- La variable pointe sur une autre variable plus ancienne.

 107    type P_Subst is record
 108       Stype : Type_Equation;
 109       Sub   : Indice_Pile_Subst;
 110       Ter   : Mot;
 111    end record;
 112 
 113    -- SUB : Si STYPE = LIE ou GELE alors SUB = base de l'environnement des variables du terme.
 114    --       Si STYPE = REF alors SUB = position absolue dans PILE_SUBST de la variable pointée.
 115    -- TER : Si STYPE = LIE alors TER = l'objet PROLOG correspondant à la valeur de la variable.
 116    --       Si STYPE = GELE alors TER = le processus gelé (qui est un objet PROLOG).
 117 
 118    Sub_Vide : constant Indice_Pile_Subst := Indice_Pile_Subst'First; -- Valeur de remplissage.

 378    procedure Sauve_Subst(Ptr : Indice_Pile_Subst) is         -- Sauvegarde la substitution indiquée.
 379    begin
 380       if Ptr < Env_Echec then
 381          Pile_Sauve(Sp_Sauve) := (Ptr, Stype(Ptr), Sub(Ptr), Ter(Ptr));
 382          if Sp_Sauve /= Pile_Sauve'Last then
 383             Sp_Sauve := Sp_Sauve + 1;
 384             if Sp_Sauve > Sp_Sauve_Max then Sp_Sauve_Max := Sp_Sauve; end if;
 385          else
 386             raise Pile_Sauve_Pleine;
 387          end if;
 388       end if;
 389    end Sauve_Subst;

 392    procedure Restaure_Subst is                               -- Restauration de la derniere substitution empilée.
 393    begin
 394       Sp_Sauve := Sp_Sauve - 1;
 395       Pile_Subst(Pile_Sauve(Sp_Sauve).Ptr_Pile_Subst) :=
 396          (Pile_Sauve(Sp_Sauve).Stype,
 397          Pile_Sauve(Sp_Sauve).Sub,
 398          Pile_Sauve(Sp_Sauve).Ter);

 473    procedure Position_Liste(Objet : in Mot; Env_Objet : in Indice_Pile_Subst;
 474          Position : out Indice_Pile_Eq; Trouve : out Boolean) is
 475    begin
 476       Trouve := False;
 477       for I in Pile_Eq'First..Sp_Eq-1 loop
 478          if Egalite_Mot(Pile_Eq(I).Terme1, Objet) and then Pile_Eq(I).Env1 = Env_Objet then
 479             Position := I;
 480             Trouve := True;
 481             return;
 482          end if;
 483       end loop;
 484    end Position_Liste;

 619          Obj : Mot := Objet;
 620          Env : Indice_Pile_Subst := Env_Objet;
 621          Sauve_Sp_Eq : Indice_Pile_Eq := Sp_Eq;
 622       begin
 623          Flagvar := Variable(Obj) or Flagvar;

 626             Ecrit(Obj, False);                                  -- FALSE pour indiquer de ne pas entourer avec des quotes.

 628             if Gele(X) then Put(Indicateur_Gele);
 629             elsif X - Sp_Subst_Initial <= Nb_Var_But then Ecrit(Nom_Global(X - Sp_Subst_Initial));
 630             else Ecrit(Cree_Variable(X));
 631             end if;
 632          else
 633             if Flagvar then
 634                -- Ici OBJ est un doublet correspondant à la valeur d'une variable. Attention aux arbres infinis.
 635                Position_Liste(Obj, Env, Position, Trouve);

 640                   Nouvelle_Liste(Obj, Env);
 641                end if;
 642             end if;
 643             if Doublet_L(Obj) then Write_Liste(Obj, Env);
 644             elsif Doublet_V(Obj) then Write_Vecteur(Obj, Env);
 645             elsif Doublet_F(Obj) then
 646                Flagvar := False;                                   -- On le met à faux, mais PREMIER(OBJ) peut etre une variable.
 647                Write_Obj(Premier(Obj), Env);
 648                Obj := Reste(Obj);
 649                Rep(Obj, Env, X, False);
 650                if Doublet(Obj) then
 651                   Flagvar := False;
 652                   Write_Obj(Obj, Env);
 653                else
 654                   Ecrit(Vecteur_Vide);
 655                end if;
 656             end if;
 657          end if;
 658          <<Restaure_Sp_Eq>>
 659             Sp_Eq := Sauve_Sp_Eq;
 660       end Write_Obj;

 663    begin -- WRITE
 664       Sp_Eq := Pile_Eq'First;                                 -- Pour les arbres infinis.
 665       Flagvar := False;                                       -- On le met à faux, mais OBJET peut etre une variable.
 666       Write_Obj(Objet, Env_Objet);
 667    end Write;

1066    -- L'unification.
1067    -----------------
1068 
1069 
1070    procedure Rep(Terme : in out Mot;                 -- L'objet PROLOG.
1071          Env : in out Indice_Pile_Subst;             -- L'environnement associé au terme.
1072          Indice : in out Indice_Pile_Subst_Etendu;   -- Adresse absolue dans PILE_SUBST ou dans PILE_EQ. (ne sert que si
1073          Recherche_Eq : in Boolean := True) is       -- TERME en sortie est une variable).
1074    begin
1075       if Variable(Terme) then
1076          if Variable_Rang(Terme) = 0 then            -- Si variable muette '_'
1077             Indice := Indice_Variable_Muette;        -- Cet indice se situe en dehors de l'étendue de PILE_SUBST.
1078             return;
1079          end if;
1080          Indice := Env + Variable_Rang(Terme);
1081          while Ref(Indice) loop Indice := Sub(Indice); end loop;
1082          if Lie(Indice) then                         -- Si INDICE pointe sur une liaison à un objet PROLOG alors
1083             Env := Sub(Indice);                      -- fait pointer ENV et TERME sur l'objet lié.
1084             Terme := Ter(Indice);
1085          end if;
1086       end if;
1087       if not Recherche_Eq then return; end if;
1088       if Doublet(Terme) then
1089          for I in Pile_Eq'First..Sp_Eq-1 loop
1090             if Egalite_Mot(Terme, Pile_Eq(I).Terme1) and Env = Pile_Eq(I).Env1 then

1125       procedure Unifier(Terme1 : in Mot; Env1 : in Indice_Pile_Subst;
1126             Terme2 : in Mot; Env2 : in Indice_Pile_Subst) is
1127          T1 : Mot := Terme1;
1128          E1 : Indice_Pile_Subst := Env1;
1129          T2 : Mot := Terme2;
1130          E2 : Indice_Pile_Subst := Env2;
1131       begin
1132          loop
1133             Rep(T1, E1, X1);
1134             Rep(T2, E2, X2);
1135             if Variable(T1) then
1136                if Variable(T2) then
1137                   -- Ici X1 et X2 sont des pointeurs dans PILE_SUBST.
1138                   if X1 = Indice_Variable_Muette
1139                         or else X2 = Indice_Variable_Muette then
1140                      return;
1141                   end if;
1142                   if X1 = X2 then return; end if;                 -- X = X toujours effacé.
1143                   if X2 < X1 then
1144                      X := X1; X1 := X2; X2 := X;                  -- Ordre chronologique (important pour éviter les bouclages).
1145                   end if;
1146                   Sauve_Subst(X2);
1147                   if Gele(X2) then Relancer(Buts_Geles, Env_Geles, X2); end if;
1148                   Pile_Subst(X2) := (Ref, X1, Ter_Vide);          -- Fait pointer la variable la plus récente vers la plus ancienne.
1149                   return;
1150                else
1151                   -- Ici T1 est une variable mais pas T2. X1 est un pointeur dans PILE_SUBST.
1152                   if X1 /= Indice_Variable_Muette then
1153                      Sauve_Subst(X1);
1154                      if Gele(X1) then Relancer(Buts_Geles, Env_Geles, X1); end if;
1155                      Pile_Subst(X1) := (Lie, E2, T2);              -- Rattache la valeur à la variable.
1156                   end if;
1157                   return;
1158                end if;
1159             elsif Variable(T2) then
1160                -- Ici T2 est une variable, mais pas T1. X2 est un pointeur dans PILE_SUBST.
1161                if X2 /= Indice_Variable_Muette then
1162                   Sauve_Subst(X2);
1163                   if Gele(X2) then Relancer(Buts_Geles, Env_Geles, X2); end if;
1164                   Pile_Subst(X2) := (Lie, E1, T1);                -- Rattache la valeur à la variable.
1165                end if;
1166                return;
1167                -- A partir d'ici aucun des termes n'est une variable.
1168             elsif E1 = E2 and then Egalite_Mot(T1, T2) then
1169                return;                                           -- Egalité stricte.
1170             elsif Atome(T1) and then Egalite_Mot(T1, T2) then
1171                return;                                           -- Si T1 et T2 atomes, l'environnement n'a pas d'importance.
1172             else
1173                if (Doublet_L(T1) and Doublet_L(T2)) or else
1174                      (Doublet_V(T1) and Doublet_V(T2)) or else
1175                      (Doublet_F(T1) and Doublet_F(T2)) then
1176                   Nouvelle_Equation(T1, E1, T2, E2);
1177                   Unifier(Premier(T1), E1, Premier(T2), E2);
1178                   T1 := Reste(T1);
1179                   T2 := Reste(T2);
1180                else
1181                   raise Pas_Unifiable;                            -- Echec, retour brutal.
1182                end if;
1183             end if;
1184          end loop;
1185       end Unifier;
1186 
1187    begin -- UNIFIABLE
1188       Sp_Eq := Pile_Eq'First;                                 -- Vide la pile des equations.
1189       Unifier(Terme1, Env1, Terme2, Env2);
1190       Sp_Eq := Pile_Eq'First;                                 -- La pile des équations ne sert plus, une fois l'unification réalisée
1191       return True;
1192    exception
1193       when Pas_Unifiable => Sp_Eq := Pile_Eq'First;           -- Meme remarque.
1194          return False;
1195    end Unifiable;

1300    procedure Effacer_Suite_De_Buts is
1301       Base_Pile_Sauve : Indice_Pile_Sauve;
1302       Env_Regle : Indice_Pile_Subst;
1303       Tete_Regle : Mot;
1304    begin
1305       if not Avancer then
1306          -- Si la suite des buts initiaux ne comportait que des regles prédéfinies, alors elles ont été effacés directement depuis
1307          -- AVANCER. BUT_COURANT va nous indiquer si les buts ont pu etre effacés.

1309          return;
1310       end if;
1311       loop
1312          loop
1313             if Egalite_Mot(Liste_Regles, Liste_Vide) then goto Backtracking; end if;
1314             Preparer_Unification(Liste_Regles, Env_Regle, Tete_Regle, Base_Pile_Sauve);
1315             while not Unifiable(But_Courant, Env_Courant, Tete_Regle, Env_Regle) loop
1316                while Sp_Sauve /= Base_Pile_Sauve loop Restaure_Subst; end loop;
1317                Sp_Subst := Env_Regle;
1318                Liste_Regles := Reste(Liste_Regles);
1319                if Egalite_Mot(Liste_Regles, Liste_Vide) then goto Backtracking; end if;
1320                Preparer_Unification(Liste_Regles, Env_Regle, Tete_Regle, Base_Pile_Sauve);
1321             end loop;
1322             if not Egalite_Mot(Reste(Liste_Regles), Liste_Vide) then
1323                Empile_Choix(Reste(Liste_Regles),
1324                   Sp_Etape,
1325                   Env_Regle,
1326                   Base_Pile_Sauve,
1327                   But_Courant,
1328                   Env_Courant,
1329                   Reste_De_Buts);
1330             end if;
1331             Pile_Subst(Env_Regle) := (Libre, Sp_Choix, Liste_Regles);   -- Attention à SP_CHOIX
1332             if not Egalite_Mot(Buts_Geles, Vecteur_Vide) then
1333                But_Courant := Buts_Geles;
1334                Env_Courant := Env_Geles;
1335                if not Egalite_Mot(Queue(Premier(Liste_Regles)), Vecteur_Vide) then
1336                   Empile_Etape(Queue(Premier(Liste_Regles)), Env_Regle);
1337                end if;
1338             else
1339                But_Courant := Queue(Premier(Liste_Regles));
1340                Env_Courant := Env_Regle;
1341             end if;
1342             exit when not Avancer;
1343          end loop;
1344          ---------------------------
1345          -- Affichage de la réponse.
1346          ---------------------------
1347          Imprimer_Reponse;
1348          if Mode_R /= Toutes_Reponses then return; end if;
1349          ---------------------
1350          -- Retour en arriere.
1351          ---------------------
1352          <<Backtracking>>
1353             if Sp_Choix = Sp_Choix_Initial then return;

1356                Sp_Etape,
1357                Sp_Subst,
1358                Base_Pile_Sauve,
1359                But_Courant,
1360                Env_Courant,
1361                Reste_De_Buts);
1362             while Sp_Sauve /= Base_Pile_Sauve loop Restaure_Subst; end loop;
1363          end if;
1364       end loop;
1365    end Effacer_Suite_De_Buts;

1448    function Eval(Expr : Mot; Env_Expr : Indice_Pile_Subst) return Mot is
1449       Erreur_Evaluation : exception;
1450       type Func1arg is (V_Abs, Moins_Unaire, Div_Unaire);
1451       type Func2arg is (Modulo, Puissance);
1452       type Funcnarg is (Plus, Moins, Mult, Div);
1453 
1454       function Evaluation(Expression : Mot; Env_Expression : Indice_Pile_Subst) return Mot;
1455 
1456       function Apply1(Func : Func1arg; Args : Mot; Env_Args : Indice_Pile_Subst) return Mot is
1457          Arg : Mot_Valeur;
1458       begin
1459          if not Vect1(Args) then raise Erreur_Evaluation; end if;
1460          Arg := Entier_Val(Evaluation(Premier(Args), Env_Args));
1461          case Func is
1462             when V_Abs        => Arg := abs Arg;
1463             when Moins_Unaire => Arg := -Arg;
1464             when Div_Unaire   => Arg := 1 / Arg;
1465          end case;
1466          return Cree_Entier(Arg);
1467       end Apply1;
1468 
1469       function Apply2(Func : Func2arg; Args : Mot; Env_Args : Indice_Pile_Subst) return Mot is
1470          Arg1, Arg2 : Mot_Valeur;
1471       begin
1472          if not Vect2(Args) then raise Erreur_Evaluation; end if;
1473          Arg1 := Entier_Val(Evaluation(Premier(Args), Env_Args));
1474          Arg2 := Entier_Val(Evaluation(Premier(Reste(Args)), Env_Args));
1475          case Func is
1476             when Modulo    => Arg1 := Arg1 mod Arg2;
1477             when Puissance => Arg1 := Arg1 ** Arg2;
1478          end case;
1479          return Cree_Entier(Arg1);
1480       end Apply2;
1481 
1482       function Apply(Func : Funcnarg; Args : Mot; Env_Args : Indice_Pile_Subst; Initial : Mot_Valeur) return Mot is
1483          Passe1  : Boolean := True;
1484          Result  : Mot_Valeur := Initial;
1485          Vectarg : Mot := Args;
1486 
1487          function Next return Mot_Valeur is
1488             V : Mot_Valeur := Entier_Val(Evaluation(Premier(Vectarg), Env_Args));
1489          begin
1490             Passe1 := False;
1491             Vectarg := Reste(Vectarg);
1492             return V;
1493          end Next;
1494 
1495       begin -- APPLY
1496          case Func is
1497             when Plus =>
1498                while Doublet_V(Vectarg) loop Result := Result + Next; end loop;
1499             when Moins =>
1500                while Doublet_V(Vectarg) loop
1501                   if Passe1 then Result := Next; else Result := Result - Next; end if;
1502                end loop;
1503             when Mult =>
1504                while Doublet_V(Vectarg) loop Result := Result * Next; end loop;
1505             when Div =>
1506                while Doublet_V(Vectarg) loop
1507                   if Passe1 then Result := Next; else Result := Result / Next; end if;
1508                end loop;
1509          end case;
1510          return Cree_Entier(Result);
1511       end Apply;
1512 
1513       function Evaluation(Expression : Mot; Env_Expression : Indice_Pile_Subst) return Mot is
1514          Func, Args : Mot;
1515          Expr : Mot := Expression;
1516          Env_Expr : Indice_Pile_Subst := Env_Expression;
1517       begin
1518          if Vect1(Expr) then Expr := Premier(Expr); end if;
1519          if Variable(Expr) then Rep(Expr, Env_Expr, X); end if;
1520          if Entier(Expr) then
1521             return Expr;
1522          elsif Doublet_F(Expr) then
1523             Func := Premier(Expr);
1524             Args := Reste(Expr);
1525             if Egalite_Mot(Func, S_Plus) then
1526                return Apply(Plus, Args, Env_Expr, 0);
1527             elsif Egalite_Mot(Func, S_Moins) then
1528                if Vect1(Args) then return Apply1(Moins_Unaire, Args, Env_Expr);
1529                else return Apply(Moins, Args, Env_Expr, 0);
1530                end if;
1531             elsif Egalite_Mot(Func, S_Etoile) then
1532                return Apply(Mult, Args, Env_Expr, 1);
1533             elsif Egalite_Mot(Func, S_Slash) then
1534                if Vect1(Args) then return Apply1(Div_Unaire, Args, Env_Expr);
1535                else return Apply(Div, Args, Env_Expr, 1);
1536                end if;
1537             elsif Egalite_Mot(Func, S_Mod) then
1538                return Apply2(Modulo, Args, Env_Expr);
1539             elsif Egalite_Mot(Func, S_Puissance) then
1540                return Apply2(Puissance, Args, Env_Expr);
1541             elsif Egalite_Mot(Func, S_Abs) then
1542                return Apply1(V_Abs, Args, Env_Expr);
1543             else
1544                raise Erreur_Evaluation;
1545             end if;
1546          else
1547             raise Erreur_Evaluation;
1548          end if;
1549       end Evaluation;
1550 
1551    begin -- EVAL
1552       return Evaluation(Expr, Env_Expr);
1553    exception
1554       when Erreur_Evaluation | Numeric_Error => return S_Err;
1555    end Eval;
