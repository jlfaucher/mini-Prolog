 214    procedure Debug is
 215       C : Character;
 216    begin
 217       loop
 218          New_Line;
 219          Put_Line("Utilitaire de mise au point.");
 220          Put_Line("----------------------------");
 221          Put_Line("(D) --> Dump doublets    (S) --> Dump symboles    (V) --> Dump variables");
 222          Put_Line("(Q) --> Retour au programme");
 223          Put("Votre choix ? ");
 224          Get(C);
 225          case C is
 226             when 'D' | 'd' => Dump_Table_Doublets;
 227             when 'S' | 's' => Dump_Table_Symb;
 228             when 'V' | 'v' => Dump_Table_Var;
 229             when 'Q' | 'q' => New_Line; return;
 230             when others    => null;
 231          end case;
 232       end loop;
 233    end;

 484    function Cree_Pname(Str : String; Print_Quote : Boolean := False) return R_Pname is
 485       -- En entrée : Une chaine, en sortie un record (long, ind_carac, flag).
 486       Longueur : Natural := Str'Length;
 487       Pname : R_Pname;
 488       Adresse : Indice_Table_Carac;
 489    begin
 490       if Longueur > Longmax_Pname then
 491          Longueur := Longmax_Pname;                         -- Si chaine trop longue alors on tronque.
 492       end if;
 493       Pname.Long := Longueur;
 494       Pname.Ind_Carac := Topcarac;
 495       Pname.Print_Quote := Print_Quote;
 496       Adresse := Pname_Libre(Longueur);
 497       Table_Carac(Adresse..Adresse+Longueur-1) := Str(Str'First..Str'First+Longueur-1);
 498       return Pname;
 499    end Cree_Pname;

 502    function Cree_Symbole(Str : String; Print_Quote : Boolean := False) return Mot is
 503       -- En entree : La représentation externe du symbole, en sortie : l'objet symbole.
 504       Indice_Symb : Indice_Table_Symb;
 505       Symb_Suiv : Symbole_Suivant;
 506       H : Hash_Code;
 507    begin
 508       if Str'Length = 0 then
 509          H := 0;
 510       else
 511          H := Character'Pos(Str(Str'First));
 512       end if;
 513       Symb_Suiv := Table_Hashcode(H);
 514       while Symb_Suiv /= -1 loop
 515          if Egal_Pname_Chaine(Table_Symb(Symb_Suiv).Pname, Str) then
 516             return (Symbole, Symb_Suiv);
 517          end if;
 518          Symb_Suiv := Table_Symb(Symb_Suiv).Hash_Link;
 519       end loop;
 520       Indice_Symb := Symbole_Libre;
 521       Table_Symb(Indice_Symb) := (Liste_Vide, Cree_Pname(Str, Print_Quote), Table_Hashcode(H));
 522       Table_Hashcode(H) := Indice_Symb;
 523       return (Symbole, Indice_Symb);
 524    end Cree_Symbole;

 527    function Cree_Variable(Str : String) return Mot is       -- Renvoie l'objet 'variable' correspondant
 528       Indice_Symb : Indice_Table_Symb;
 529       Indice_Var        : Indice_Table_Var;
 530    begin
 531       if Str = "_" then return (Variable, 0); end if;       -- Variable muette pour optimiser les unifications.
 532       Indice_Symb := Cree_Symbole(Str).Val;
 533       if Topvar /= Table_Var'First then                       -- Si table non vide
 534          for I in Table_Var'First..Topvar-1 loop
 535             if Indice_Symb = Table_Var(I) then
 536                return (Variable, I);
 537             end if;
 538          end loop;
 539       end if;
 540       Table_Var(Topvar) := Indice_Symb;
 541       Indice_Var := Topvar;
 542       if Topvar = Table_Var'Last then raise Table_Var_Pleine; end if;
 543       Topvar := Topvar + 1;
 544       return (Variable, Indice_Var);
 545    end Cree_Variable;

 593    function Concatene(Obj1, Obj2 : Mot) return Mot is       -- Concatene le vecteur, la fonction ou la liste OBJ1 avec OBJ2.
 594       Dernier_Doublet : Mot := Obj1;                        -- Le résultat sera du meme type que OBJ1 (vecteur, fonction ou liste)
 595    begin
 596       if not Doublet(Obj1) then return Obj1;
 597       else
 598          while Doublet(Reste(Dernier_Doublet)) loop
 599             Dernier_Doublet := Reste(Dernier_Doublet);
 600          end loop;
 601          Rest(Dernier_Doublet.Val) := Obj2;                 -- Fait pointer le dernier doublet sur OBJ2.
 602          return Obj1;
 603       end if;
 604    end Concatene;

 643    function Vecteur_Liste(Obj : Mot) return Mot is          -- Transforme physiquement un vecteur en liste.
 644       Objet : Mot := Obj;
 645    begin
 646       if not Doublet_V(Obj) then return Obj;
 647       else
 648          while Doublet_V(Reste(Objet)) loop
 649             Rest(Objet.Val).T := Doublet_L;                 -- Modifie physiquement le type du reste de OBJET.
 650             Objet := Reste(Objet);                                  -- Maintenant c'est un doublet liste.
 651          end loop;
 652          return (Doublet_L, Obj.Val);                       -- Ne pas oublier de modifier le type du pointeur initial.
 653       end if;
 654    end Vecteur_Liste;
