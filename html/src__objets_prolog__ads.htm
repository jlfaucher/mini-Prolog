 137    ---------------------------------------------------------------
 138    -- Déclaration des types possibles du contenu d'un mot mémoire.
 139    -- ATTENTION !!! Ne pas changer l'ordre des types.
 140    ---------------------------------------------------------------
 141    type Contenu_Memoire is (Libre,                  -- Le mot est libre
 142                             Variable,               -- Le mot contient un numéro de variable
 143                             Symbole,                -- Le mot contient un indice faisant référence à la table des symboles
 144                             Entier,                 -- Le mot contient un entier
 145                             Doublet_V,              -- Le mot contient une adresse pointant sur un doublet-vecteur
 146                             Doublet_F,              -- Le mot contient une adresse pointant sur un doublet-fonction
 147                             Doublet_L);             -- Le mot contient une adresse pointant sur un doublet-liste

 155       Val    : Mot_Valeur;                          -- Le contenu du mot mémoire
 156    end record;
